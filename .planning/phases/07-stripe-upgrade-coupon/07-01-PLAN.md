    ```typescript
    // Check if user has an existing Analyst purchase for upgrade credit
    let upgradeCouponId: string | undefined
    if (productId === 'strategist_annual' && STRIPE_UPGRADE_COUPON_ID) {
      const analystPurchases = await sql`
        SELECT id FROM purchases
        WHERE user_id = ${userId}
          AND product_id = 'analyst_annual'
          AND status = 'completed'
        LIMIT 1
      `
      if (analystPurchases.length > 0) {
        upgradeCouponId = STRIPE_UPGRADE_COUPON_ID
        console.log('[Checkout] Analyst upgrade detected — applying $4,999 coupon')
      }
    }
    ```

    **Step 3: Apply coupon to checkout session.** Modify the subscription checkout session creation (lines 110-118) to include the `discounts` parameter when an upgrade coupon applies:

    ```typescript
    const checkoutSession = await getStripe().checkout.sessions.create({
      customer: customerId,
      mode: 'subscription' as const,
      payment_method_types: ['card' as const],
      line_items: [lineItem],
      ...(upgradeCouponId ? { discounts: [{ coupon: upgradeCouponId }] } : {}),
      success_url: `${baseUrl}/dashboard?checkout=success`,
      cancel_url: `${baseUrl}/dashboard?checkout=canceled`,
      metadata: { user_id: userId, product_id: productId },
    })
    ```

    **Important:** The `discounts` array uses `coupon` (not `promotion_code`). The Stripe Checkout Sessions API accepts `discounts` only when `allow_promotion_codes` is not set — since we do not set `allow_promotion_codes` in our current code, this is safe.
